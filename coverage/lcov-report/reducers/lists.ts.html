import {
  Action,
  APPEND_MESSAGE,
  AppendMessage,
  LOAD_LIST,
  LoadList
} from '../actions/list';
import { Message } from './drafts';
 
export type List = Message[];
 
export type State = Map<string, List>;

export function loadListCase(state: State, action: LoadList): State {
  return action.list.reduce(
    (result: State, message: Message) => {
      const { subjectId } = message;
      const list = result.get(subjectId) || [];
 
      return result.set(subjectId, list.concat(message));
    },
    new Map(state)
  );
}
 
export function appendMessageCase(state: State, action: AppendMessage): State {
  const {
    message,
    message: { subjectId }
  } = action;
  const result: State = new Map(state);
  const list = result.get(subjectId) || [];
 
  return result.set(subjectId, list.concat(message));
}
 
export default function(state: State = new Map(), action: Action): State {
  switch (action.type) {
    case APPEND_MESSAGE:
      return appendMessageCase(state, action);
    case LOAD_LIST:
      return loadListCase(state, action);
    default:
      return state;
  }
}
 
export function getList(state: State, subjectId: string = null): List {
  return state.get(subjectId);
}
 
export function hasList(state: State, subjectId: string = null): boolean {
  return state.has(subjectId);
}
