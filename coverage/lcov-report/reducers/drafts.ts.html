import {
  Action,
  CHANGE,
  Change,
  CLEAN,
  Clean,
  CREATE,
  Create
} from '../actions/drafts';

export interface Draft {
  subjectId: string;
  text: string;
}
 
export interface Message extends Draft {
  id: string;
  date: Date;
}

export type State = Map<string, Draft>;

const defaultState: State = new Map([
  [null, {
    subjectId: null,
    text: ''
  }]
]);

export function createCase(state: State, action: Create): State {
  const { subjectId } = action;
  const result = new Map(state);
 
  return result.set(subjectId, {
    subjectId,
    text: ''
  });
}
 
export function changeCase(state: State, action: Change): State {
  const {
    subjectId,
    text
  } = action;
  const result = new Map(state);

  return result.set(subjectId, {
    ...result.get(subjectId),
    text
  });
}
 
export function cleanCase(state: State, action: Clean): State {
  const { subjectId } = action;
  const result = new Map(state);
 
  if (subjectId === null) {
    result.set(subjectId, {
      subjectId: null,
      text: ''
    });
  } else {
    result.delete(subjectId);
  }
 
  return result;
}
 
export default function(state: State = defaultState, action: Action): State {
  switch (action.type) {
    case CREATE:
      return createCase(state, action);
    case CHANGE:
      return changeCase(state, action);
    case CLEAN:
      return cleanCase(state, action);
    default:
      return state;
  }
}
 
export function getDraft(state: State, subjectId: string = null): Draft {
  return state.get(subjectId);
}
 
export function hasDraft(state: State, subjectId: string = null): boolean {
  return state.has(subjectId);
}
 
export function isValid(draft: Draft): boolean {
  return !!draft.text;
}
