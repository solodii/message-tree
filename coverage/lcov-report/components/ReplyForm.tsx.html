import * as React from 'react';
import { Draft } from '../reducers/drafts';
 
export interface Props {
  draft: Readonly<Partial<Draft>>;
  isValid: boolean;
  onChange: (e: React.ChangeEvent<HTMLInputElement>) => void;
  onCancel: (e: React.MouseEvent<HTMLButtonElement>) => void;
  onSubmit: (e: React.MouseEvent<HTMLButtonElement>) => void;
}
 
export default function ReplyForm(props: Props) {
  const {
    draft: { text },
    isValid,
    onChange,
    onCancel,
    onSubmit
  } = props;
 
  return (
    <div className="form">
      <input
        type="text"
        autoFocus
        placeholder="Type your message here..."
        value={ text }
        onChange={ onChange }
      />
      <button
        className="cancel"
        onClick={ onCancel }
      >Cancel</button>
      <button
        className="submit"
        disabled={ !isValid }
        onClick={ onSubmit }
      >Submit</button>
    </div>
  );
}
