import * as React from 'react';
import { shallow } from 'enzyme';
import * as dateformat from 'dateformat';
import { Draft } from '../../reducers/drafts';
import ReplyForm, { Props } from '../ReplyForm';
 
function setup() {
  const props: Props = {
    draft: {
      text: ''
    },
    isValid: false,
    onChange: jest.fn(),
    onSubmit: jest.fn(),
    onCancel: jest.fn()
  };
  const wrapper = shallow(
    <ReplyForm { ...props } />
  );
  
  return {
    props,
    wrapper
  };
}
 
describe('ReplyForm component', () => {
  test('should render self and subcomponents', () => {
    const {
      props: {
        isValid,
        draft: { text },
        onChange,
        onCancel,
        onSubmit
      },
      wrapper
    } = setup();
 
    expect(wrapper.is('.form')).toBeTruthy();
 
    const input = wrapper.find('input');
 
    expect(input.length).toEqual(1);
    expect(input.prop('type')).toEqual('text');
    expect(input.prop('autoFocus')).toBeTruthy();
    expect(input.prop('placeholder')).toEqual('Type your message here...');
    expect(input.prop('value')).toEqual(text);
    expect(input.prop('onChange')).toBe(onChange);
 
    const buttons = wrapper.find('button');
 
    expect(buttons.length).toEqual(2);
 
    const cancel = buttons.filter('.cancel');
 
    expect(cancel.length).toEqual(1);
    expect(cancel.text()).toEqual('Cancel');
    expect(cancel.prop('onClick')).toBe(onCancel);
 
    const submit = buttons.filter('.submit');
 
    expect(submit.length).toEqual(1);
    expect(submit.text()).toEqual('Submit');
    expect(submit.prop('disabled')).toEqual(!isValid);
    expect(submit.prop('onClick')).toBe(onSubmit);
  });
 
  test('should response on props change', () => {
    const {
      wrapper,
      props
    } = setup();
 
    expect(
      wrapper.find('input').prop('value')
    ).toEqual(props.draft.text);
    expect(
      wrapper.find('.submit').prop('disabled')
    ).toEqual(!props.isValid);
 
    const text = Math.random().toString();
    const nextProps = {
      draft: { text },
      isValid: true
    };
 
    wrapper.setProps(nextProps);
 
    expect(
      wrapper.find('input').prop('value')
    ).toEqual(text);
    expect(
      wrapper.find('.submit').prop('disabled')
    ).toEqual(!nextProps.isValid);
  });
 
  test('should invoke callbacks from props', () => {
    const {
      wrapper,
      props: {
        onChange,
        onCancel,
        onSubmit
      }
    } = setup();
 
    expect(onChange).not.toHaveBeenCalled();
    expect(onCancel).not.toHaveBeenCalled();
    expect(onSubmit).not.toHaveBeenCalled();
 
    wrapper.find('input').props().onChange({} as any);
    wrapper.find('.cancel').props().onClick({} as any);
    wrapper.find('.submit').props().onClick({} as any);
 
    expect(onChange).toHaveBeenCalledTimes(1);
    expect(onCancel).toHaveBeenCalledTimes(1);
    expect(onSubmit).toHaveBeenCalledTimes(1);
  });
});
