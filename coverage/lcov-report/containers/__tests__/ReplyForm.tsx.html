import {
  change,
  clean,
  submit
} from '../../actions/drafts';
import rootReducer from '../../reducers/__mocks__';
import * as fromReducers from '../../reducers';
import * as fromDrafts from '../../reducers/drafts';
import ReplyFormComponent, {
  Props as ComponentProps
} from '../../components/ReplyForm';
import {
  mapStateToProps,
  mapDispatchToProps,
  mergeProps
} from '../ReplyForm';
 
describe('ReplyForm container', () => {
  const state = rootReducer();
  const subjectId = '1';
  const dispatch = jest.fn();
  const mergedProps: ComponentProps = mergeProps(
    mapStateToProps(state, { subjectId }),
    mapDispatchToProps(dispatch)
  );
 
  afterEach(() => dispatch.mockClear());
 
  test('merged props should be correct', () => {
    const draft = fromReducers.getDraft(state, subjectId);
    const isValid = fromDrafts.isValid(draft);
 
    expect(mergedProps.draft).toEqual(draft);
    expect(mergedProps.isValid).toEqual(isValid);
    expect(mergedProps.onChange).toBeInstanceOf(Function);
    expect(mergedProps.onCancel).toBeInstanceOf(Function);
    expect(mergedProps.onSubmit).toBeInstanceOf(Function);
  });
 
  test(`merged props' callbacks should dispatch actions`, () => {
    const value = Math.random().toString();
    
    expect(dispatch).not.toBeCalled();
 
    mergedProps.onChange({
      target: { value }
    } as any);
    mergedProps.onCancel({} as any);
    mergedProps.onSubmit({} as any);
 
    expect(dispatch.mock.calls).toEqual([
      [change(value, subjectId)],
      [clean(subjectId)],
      [submit(subjectId)]
    ]);
  });
});
